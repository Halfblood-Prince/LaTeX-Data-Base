\makeatletter \define@key{mykeys}{name}{\def\imgname{#1}} \define@key{mykeys}{width}{\def\imgwidth{#1}} \define@key{mykeys}{caption}{\def\imgcaption{#1}} \define@key{mykeys}{label}{\def\imglabel{#1}} \newcommand{\img}[1]{%
    \setkeys{mykeys}{#1}%
\begin{figure}[h] \centering \includegraphics[width= \imgwidth, keepaspectratio]{\imgname} \captionsetup{justification=centering} \caption{\imgcaption} \label{\imglabel} \end{figure} } \makeatother

\makeatletter \define@key{mykeys}{e}{\def\equation{#1}} \define@key{mykeys}{label}{\def\eqnlabel{#1}} \newcommand{\eq}[1]{%
    \setkeys{mykeys}{#1}%
\begin{equation} \label{#2} #1 \end{equation} } \makeatother

\def \p#1{  \noindent #1  }  \def \pg#1{  \noindent #1 \\  }  

\def \pb{  \vfill \pagebreak  }  \def \imgrow#1#2#3#4#5#6#7#8{  \begin{figure}[h] \centering \begin{subfigure}{0.45\textwidth} \centering \includegraphics[width=\linewidth]{#1} \captionsetup{justification=centering} \caption{#2} \label{#3} \end{subfigure} \begin{subfigure}{0.45\textwidth} \centering \includegraphics[width=\linewidth]{#4} \captionsetup{justification=centering} \caption{#5} \label{#6} \end{subfigure} \captionsetup{justification=centering} \caption{#7} \label{#8} \end{figure}  }
